input: alpha, g_1, g_2, vektoren mit dot(g_1, g_2)=0 |g_1|=|g_2|=1
V = g_1*transpose(g_1) + g_2*transpose(g_2)
W = g_1*transpose(g_2) - g_2*transpose(g_1)
R = I_n + (cos(alpha) - 1) * V - sin(alpha)W
